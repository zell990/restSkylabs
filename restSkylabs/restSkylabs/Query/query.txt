/*guadagno di capitale medio per ogni categoria
lavorativa*/
SELECT t2.name as categoria_lavorativa,AVG(capital_gain - capital_loss) as capitale_medio
FROM records as t1
LEFT JOIN occupations as t2 ON t1.occupation_id = t2.id
GROUP BY t2.name
